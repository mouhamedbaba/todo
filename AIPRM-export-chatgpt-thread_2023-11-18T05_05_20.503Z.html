    // src/App.js
import React, { useState, useEffect } from 'react';
import axios from 'axios';

function App() {
  const [tasks, setTasks] = useState([]);
  const [newTask, setNewTask] = useState('');

  useEffect(() => {
    axios.get('http://localhost:8000/api/tasks/')
      .then(response => setTasks(response.data))
      .catch(error => console.error('Error fetching tasks:', error));
  }, []);

  const addTask = () => {
    axios.post('http://localhost:8000/api/tasks/', { title: newTask, completed: false })
      .then(response => setTasks([...tasks, response.data]))
      .catch(error => console.error('Error adding task:', error));
  };

  const toggleTask = (taskId) => {
    axios.patch(`http://localhost:8000/api/tasks/${taskId}/`, { completed: true })
      .then(response => {
        const updatedTasks = tasks.map(task => (task.id === taskId ? response.data : task));
        setTasks(updatedTasks);
      })
      .catch(error => console.error('Error updating task:', error));
  };

  return (
    <div>
      <h1>Todo List</h1>
      <ul>
        {tasks.map(task => (
          <li key={task.id} onClick={() => toggleTask(task.id)}>
            {task.title} {task.completed && '(completed)'}
          </li>
        ))}
      </ul>
      <div>
        <input type="text" value={newTask} onChange={(e) => setNewTask(e.target.value)} />
        <button onClick={addTask}>Add Task</button>
      </div>
    </div>
  );
}

export default App;
